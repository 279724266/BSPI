        <h1>MQ-2 烟雾/燃气监测（纯GPIO版）</h1>

        <div class="pin-info">MQ2 DO引脚 → 树莓派物理16号引脚</div>
        <div class="update-time" id="updateTime">更新时间：--:--:--</div>
    </div>

    <script>
        // 树莓派IP（本地用localhost，其他设备替换为树莓派实际IP）
        const piIp = "localhost";
        const port = 3000;
        const apiUrl = `http://${piIp}:${port}/mq2/status`;

        // 每秒刷新一次状态（实时监测）
        setInterval(() => getMQ2Status(), 1000);
        // 初始化立即读取
        getMQ2Status();

        // 读取传感器状态
        async function getMQ2Status() {
            try {
                const res = await fetch(apiUrl);
                const data = await res.json();
                if (data.status === "success") {
                    const statusDom = document.getElementById("mq2Status");
                    statusDom.textContent = data.statusText;
                    statusDom.className = `status-box ${data.level}`;
                    // 更新时间
                    const now = new Date();
                    document.getElementById("updateTime").textContent = `更新时间：${now.toLocaleTimeString()}`;
                }
            } catch (err) {
                const statusDom = document.getElementById("mq2Status");
                statusDom.textContent = "❌ 传感器读取失败，请检查接线";
                statusDom.className = "status-box danger";
            }
        }
    </script>
